150 display.print(label);
151
152 // Print the value on the right
153 display.setCursor(SCREEN_WIDTH - fullValueWidth, row);
154 display.print(fullValue);
155 };
156
157 int baseRow = 17; // Start below the headline

159
160 // Update only the readings area without clearing it
161 printAligned("CO2:", co2, "ppm", baseRow);
162 printAligned("T (SCD30):", temperatureSCD, "C", baseRow + rowSpacing);
163 printAligned("T (BMP280):", temperatureBMP, "C", baseRow + 2 * rowSpacing);
164 printAligned("Humidity:", humidity, "%", baseRow + 3 * rowSpacing);
